Chains and playback

Below you will find working example(s) and complete, ready to use HTML code for each example.

Remember: always include the correct path to jQuery library and the parallax plugin:

 <!-- PROVIDE THE CORRECT PATH TO .js FILES -->
<script type="text/javascript" src="/path/to/jquery-latest.js"></script> 
<script type="text/javascript" src="/path/to/jquery.real-parallax.js"></script>

Chain of actions and playback

    Mouse emulation

  • mouse to (70%,90%)
  • mouse to random position

HTML code

Complete html code of example above (show / hide)

<!DOCTYPE html>
<html xmlns="http://www.w3.org/1999/xhtml">
    <head>
        <meta http-equiv="Content-Type" content="text/html; charset=UTF-8" />
        <meta name="description" content="Real Parallax effect jQuery plugin" />
        <title>Parallax</title>
        
        <style type="text/css">
        .box{
            position: relative;
            overflow:hidden;
            padding: 0px;   
            margin: 0px;
            border:1px solid #ccc;          
        }
        </style>
        
        <!-- PROVIDE THE CORRECT PATH TO .js FILES -->
        <script type="text/javascript" src="/path/to/jquery-latest.js"></script> 
        <script type="text/javascript" src="/path/to/jquery.real-parallax.js"></script> 
            
        <script type="text/javascript">
            $(document).ready(function() 
                { 
                    //START PARALLAX                     
                    $('#parallaxID').parallax({width: 750,  height: 370,focusZ:10, sensitivityY:-0.7, sensitivityX:-0.9, easingCoefficient:20}); 
                                
                    //add some actions to chain
                    $('#parallaxID').parallax('addToChain','mouseTo', {x:0, y:0});
                    $('#parallaxID').parallax('addToChain','mouseTo', {x:390, y:270});
                    $('#parallaxID').parallax('addToChain','mouseTo', {x:390, y:0});
                    $('#parallaxID').parallax('addToChain','mouseTo', {x:0, y:270});
                    $('#parallaxID').parallax('addToChain','mouseToRandom');
                } 
            );
            
            function mouseSwitch(){
                var mouseEnabled=document.getElementById("mouseenable").checked;
                document.getElementById("mousemove").disabled=!mouseEnabled;                
                document.getElementById("mouseclick").disabled=!mouseEnabled;               
                $('#parallaxID').parallax('option','enableMouse',mouseEnabled);                     
            }
            
            function mouseSwitchMode(){
                var mouseClick=document.getElementById("mouseclick").checked;                                       
                $('#parallaxID').parallax('option','activateOnClick',mouseClick);                           
            }
        
        </script>   
        
    </head>
    
    <body>   
            
        <!-- PARALLAX -->
        <div id="parallaxID" class="box">        
            <div style="left:-1px;top:-1px;" data-lock="yes"><img src="img/planets/stars.jpg"/></div>
            <div id="planet1" style="left:50%;top:20%;" data-z="70"><img src="img/planets/planet1.png"/></div>
            <div id="sun" style="left:40%;top:30%;" data-z="0"><img src="img/planets/sun.png"/></div>
            <div id="planet2" style="left:25%;top:30%;" data-z="-70"><img src="img/planets/planet2.png"/></div>
        </div>
                
        <!-- CONTROLS -->
        <table>               
          <tr>                  
            <td>
                <ul><h5>Mouse emulation</h5>
                <li><a href="#" onmousedown="$('#parallaxID').parallax('mouseTo', {x:0, y:0})" onClick="return false"> mouse to (0px,0px) </a></li>
                <li><a href="#" onmousedown="$('#parallaxID').parallax('mouseToRatio', {x:0.7, y:0.9})" onClick="return false"> mouse to (70%,90%)</a></li>
                <li><a href="#" onmousedown="$('#parallaxID').parallax('mouseToRandom')" onClick="return false"> mouse to random position </a></li>
                <li><a href="#" onmousedown="$('#parallaxID').parallax('reset')" onClick="return false"> mouse to center </a></li>
                </ul>
            </td>
            <td>
                <ul><h5>Layers manipulation</h5>
                <li><a href="#" onmousedown="$('#parallaxID').parallax('layer','sun',{x:Math.random()*700, y:Math.random()*300})" onClick="return false">replace the SUN layer</a></li>
                <li><a href="#" onmousedown="$('#parallaxID').parallax('centerTo', {id:'planet1', align:'tr', offx:350, offy:-30})" onClick="return false">blue planet to the right</a></li>
                <li><a href="#" onmousedown="$('#parallaxID').parallax('centerTo', {id:'planet2', align:'bl', offx:-100, offy:-30})" onClick="return false">sandy planet up</a></li>
                <li>&nbsp;</li>
                </ul>
            </td>
          </tr>
                 
          <tr>
            <td>
                <form>
                    <input type="checkbox" id="mouseenable" checked onclick="mouseSwitch()"/><label for="mouseenable"> enable mouse</label><br>
                    <input type="radio" name="group1" value="1" id="mousemove" checked onchange="mouseSwitchMode()"/><label for="mousemove"> on mouse move</label><br>
                    <input type="radio" name="group1" value="2" id="mouseclick" onchange="mouseSwitchMode()"/><label for="mouseclick"> on mouse click</label>
                </form>
            </td>
            <td>
                <ul><h5>Auto playback</h5>                      
                <li><a id="play" class="button" href="#" onmousedown="$('#parallaxID').parallax('startChain', true)" onClick="return false"> start auto play </a></li>
                <li><a id="stop" class="button" href="#" onmousedown="$('#parallaxID').parallax('stopChain')" onClick="return false"> stop </a></li>
                <li><a id="pause" class="button" href="#" onmousedown="$('#parallaxID').parallax('triggerPause')" onClick="return false"> trigger pause </a></li>
                <li><a id="pause3s" class="button" href="#" onmousedown="$('#parallaxID').parallax('pause',3000)" onClick="return false"> pause 3s </a></li>
                </ul>
            </td>
            
          </tr>               
        </table>
        
    </body>
</html>
