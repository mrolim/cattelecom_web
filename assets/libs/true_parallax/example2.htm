Content of the layers

Below you will find working example(s) and complete, ready to use HTML code for each example.

Remember: always include the correct path to jQuery library and the parallax plugin:

 <!-- PROVIDE THE CORRECT PATH TO .js FILES -->
<script type="text/javascript" src="/path/to/jquery-latest.js"></script> 
<script type="text/javascript" src="/path/to/jquery.real-parallax.js"></script>

locked background layer

HTML code

Complete html code of example above (show / hide)


Note: you can use any styles of the layers DIV, even background image.

<!DOCTYPE html>
<html xmlns="http://www.w3.org/1999/xhtml">
    <head>
        <meta http-equiv="Content-Type" content="text/html; charset=UTF-8" />
        <meta name="description" content="Real Parallax effect jQuery plugin" />
        <title>Parallax</title>
        
        <style type="text/css">
        #parallaxID {
            position: relative;
            overflow:hidden;
            padding: 0px;   
            margin: 0px;
            border:1px solid #ccc;          
        }
        </style>
        
        <!-- PROVIDE THE CORRECT PATH TO .js FILES -->
        <script type="text/javascript" src="/path/to/jquery-latest.js"></script> 
        <script type="text/javascript" src="/path/to/jquery.real-parallax.js"></script> 
        
        <script type="text/javascript">
            $(document).ready(function() 
                { 
                    $("#parallaxID").parallax({width: 640,  height: 280});
                } 
            );
        </script>   
        
    </head>
    
    <body>    
        <div id="parallaxID">        
            <div style="left:-1px;top:-1px;background-image: url(img/planets/stars.jpg);background-repeat:no-repeat; height: 282px; width: 642px;color:#fff;" data-lock="yes">locked background layer</div>
            <div style="left:50%;top:20%;"><img src="img/planets/planet1.png"/></div>
            <div style="left:40%;top:30%;"><img src="img/planets/sun.png"/></div>
            <div style="left:25%;top:30%;"><img src="img/planets/planet2.png"/></div>
        </div>
    </body>
</html>

<!DOCTYPE html>
<html xmlns="http://www.w3.org/1999/xhtml">
    <head>
        <meta http-equiv="Content-Type" content="text/html; charset=UTF-8" />
        <meta name="description" content="Real Parallax effect jQuery plugin" />
        <title>Parallax</title>
        
        <style type="text/css">
        #parallaxID {
            position: relative;
            overflow:hidden;
            padding: 0px;   
            margin: 0px;
            border:1px solid #ccc;          
        }
        </style>
        
        <!-- PROVIDE THE CORRECT PATH TO .js FILES -->
        <script type="text/javascript" src="/path/to/jquery-latest.js"></script> 
        <script type="text/javascript" src="/path/to/jquery.real-parallax.js"></script> 
        <script type="text/javascript" src="/path/to/swfobject.js"></script>
        
        <script type="text/javascript">
            var flashvars = {};
            var params = {
                menu: "false",
                scale: "noScale",
                allowFullscreen: "false",
                allowScriptAccess: "always",
                wmode: "transparent",
                bgcolor: "#ffffff"          
            };
            var attributes = {
                id:"preview"
            };
            swfobject.embedSWF("swf/abstract.swf", "altContent", "400px", "150px", "10.0.0", "swf/expressInstall.swf", flashvars, params, attributes);
        </script>   
        
        <script type="text/javascript">
            $(document).ready(function() 
                { 
                    $("#parallaxID").parallax({width: 640,  height: 280});
                } 
            );
        </script>   
        
    </head>
    
    <body>    
        <div id="parallaxID">        
            
            <!-- layer containing flash -->             
            <div class="" style="left:118px;top:30px;width:400px;height:150px;">                
                <div id="altContent">
                    <h1>preview</h1>
                    <p>Alternative flash content</p>
                    <p><a href="http://www.adobe.com/go/getflashplayer"><img 
                        src="http://www.adobe.com/images/shared/download_buttons/get_flash_player.gif" 
                        alt="Get Adobe Flash player" /></a></p>
                </div>              
            </div>
            
            <!-- layer containing menu -->  
            <div style="left:390px;top:40%;">
                <div class="thumb_img border-box">
                    <h4>Menu</h4>                       
                    <ul>            
                    <li><a href="http://activeden.net/">Activeden</a></li>
                    <li><a href="http://codecanyon.net/">Codecanyon</a></li>                    
                    </ul>
                </div>
            </div>
            
            <!-- layer containing only one image -->
            <div style="left:20%;top:50%;"><img src="img/planets/sun.png"/></div>
            
        </div>
    </body>
</html>
